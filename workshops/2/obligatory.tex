\begin{enumerate}
    \item Даны векторы:
    \[
    \vec a =
    \begin{pmatrix}
    	1 \\ 2 \\ 3
    \end{pmatrix},
    \quad
    \vec b =
    \begin{pmatrix}
    	5 \\ 5 \\ 3
    \end{pmatrix},
    \quad
    \vec c = 
	\begin{pmatrix}
    	8 \\ 5 \\ -2
    \end{pmatrix}.
   \]
   Найдите:
   \begin{center}
   \begin{enumerate*}[itemjoin=\hspace{2cm}]
       \item $\vec d = 3 \vec a -2 \vec b + \vec c$
       \item $\vec e = 2 \vec a + \vec b - \vec c$
   \end{enumerate*}
   \end{center}
   
   \item Докажите, что система векторов
	    $(1\ 2\ 3)^T$,
	    $(3\ 2\ 1)^T$,
	    $(1\ 0\ 1)^T$
	    является базисом в пространстве, и найдите координаты векторов 
	    $(5\ 4\ 2)^T$,
	    $(5\ 2\ 3)^T$
	    в этом базисе.
	    
	\item Известно, что $\vec a$, $\vec b$ и $\vec c$ --- некомпланарные векторы. Определите, компланарны ли векторы $\vec l$, $\vec m$ и $\vec n$, если:
		\begin{tasks}(2)
	       \task $\vec l = \vec a + \vec b$\\
	      	 	 $\vec m = 2 \vec a + \vec b - \vec c$\\
	      	 	 $\vec n = 3 \vec a -4 \vec b + \vec c$
	       \task $\vec l = 3 \vec a + \vec b - \vec c$\\
	      	 	 $\vec m =  \vec a + \vec b + \vec c$\\
	      	 	 $\vec n = 5 \vec a +3 \vec b + \vec c$
	    \end{tasks}
	    
	\item Даны векторы
	$$\vec a = \begin{pmatrix}
	    	1 \\ 1
	    \end{pmatrix},\quad
	   \vec b = \begin{pmatrix}
	    	-1 \\ 4
	    \end{pmatrix},\quad
	    \vec c = \begin{pmatrix}
	    	2 \\ -1
	    \end{pmatrix}.$$
	  
	 Найти $\alpha$ и $\beta$ такие, что
	 $$\alpha \vec a + \beta \vec b + \c = \vec 0.$$
	    

\end{enumerate}
