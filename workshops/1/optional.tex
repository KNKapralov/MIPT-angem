\def \ProveTransposeTask {%
    \item Докажите тождества:
    \begin{tasks}(2)
        \task $(\alpha A)^T = \alpha A^T$
        \task $(A + B)^T = A^T + B^T$
        \task $(AB)^T = B^T A^T$
        \task $(A_1 A_2 \dots A_n)^T = A_n^T \dots A_2^T A_1^T$
    \end{tasks}
}
